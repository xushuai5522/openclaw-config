from rembg import remove

# 1. 抠图
img = Image.open('input.jpg')
output = remove(img)
output.save('rmbg.png')

# 2. 叠加白底
white_bg = Image.new('RGBA', (800, 800), (255, 255, 255, 255))
white_bg.paste(output, ((800-output.width)//2, (800-output.height)//2))
white_bg.convert('RGB').save('output.jpg', 'JPEG', quality=95)
